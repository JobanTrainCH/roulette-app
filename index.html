const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const colors = ['#FFADAD', '#FFD6A5', '#FDFFB6', '#CAFFBF', '#9BF6FF', '#A0C4FF']; let currentRotation = 0; function draw() { const items = Array.from(document.querySelectorAll('.item')).map(i => i.value).filter(v => v !== ""); const num = items.length; const arc = (Math.PI * 2) / num; ctx.clearRect(0, 0, 300, 300); items.forEach((text, i) => { const angle = currentRotation + i * arc; ctx.beginPath(); ctx.fillStyle = colors[i % colors.length]; ctx.moveTo(150, 150); ctx.arc(150, 150, 140, angle, angle + arc); ctx.fill(); // テキスト描画 ctx.save(); ctx.translate(150, 150); ctx.rotate(angle + arc / 2); ctx.textAlign = "right"; ctx.fillStyle = "#333"; ctx.font = "bold 16px sans-serif"; ctx.fillText(text, 130, 10); ctx.restore(); }); } function spin() { const extraSpin = Math.random() * 2000 + 2000; // 3〜5回転くらい const start = performance.now(); const duration = 3000; // 3秒間 function animate(time) { const progress = Math.min((time - start) / duration, 1); const easeOut = 1 - Math.pow(1 - progress, 3); // 滑らかに止まる currentRotation += (extraSpin / 100) * (1 - easeOut); draw(); if (progress < 1) requestAnimationFrame(animate); } requestAnimationFrame(animate); } // 入力変更時に再描画 document.getElementById('inputContainer').addEventListener('input', draw); draw();
